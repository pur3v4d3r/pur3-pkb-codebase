## 7. First Workflow

### 7.1 Create Workflow Execution Script

```bash
cd ~/PKB-System/spes

# Create simple_workflow.py
cat > simple_workflow.py << 'EOF'
#!/usr/bin/env python3
"""
Simple SPES Workflow Example
Demonstrates component composition and LLM execution
"""

import os
from pathlib import Path
from dotenv import load_dotenv
from anthropic import Anthropic

# Load environment
load_dotenv()

def load_component(component_path: str) -> str:
    """Load a component file and extract content (skip YAML frontmatter)."""
    path = Path(component_path)
    
    if not path.exists():
        raise FileNotFoundError(f"Component not found: {component_path}")
    
    content = path.read_text()
    
    # Skip YAML frontmatter
    if content.startswith('---'):
        parts = content.split('---', 2)
        if len(parts) >= 3:
            content = parts[2].strip()
    
    return content

def compose_prompt(components: list, user_input: str) -> str:
    """Compose components into a single prompt."""
    sections = []
    
    for component in components:
        sections.append(load_component(component))
    
    # Add user input
    sections.append(f"\n\n# User Request\n\n{user_input}")
    
    return "\n\n".join(sections)

def execute_workflow(prompt: str, llm_provider: str = "claude") -> str:
    """Execute prompt with specified LLM."""
    
    if llm_provider == "claude":
        client = Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))
        
        response = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=4096,
            messages=[
                {"role": "user", "content": prompt}
            ]
        )
        
        return response.content[0].text
    
    else:
        raise ValueError(f"Unsupported LLM provider: {llm_provider}")

def main():
    """Run the simple workflow."""
    
    print("=" * 60)
    print("SPES Simple Workflow - Code Generation")
    print("=" * 60)
    print()
    
    # Define components to use
    components = [
        "components/core/personas/persona-python-expert-v1.0.0.md",
        "components/core/instructions/instruction-code-generation-v1.0.0.md",
        "components/core/formats/format-code-with-explanation-v1.0.0.md"
    ]
    
    # User input
    user_request = """
    Create a Python function that calculates the Fibonacci sequence up to n terms.
    The function should be efficient and handle edge cases.
    """
    
    print("Components:")
    for comp in components:
        print(f"  - {Path(comp).name}")
    print()
    
    print("User Request:")
    print(user_request.strip())
    print()
    print("-" * 60)
    print("Composing prompt and executing workflow...")
    print("-" * 60)
    print()
    
    # Compose prompt
    prompt = compose_prompt(components, user_request)
    
    # Execute
    result = execute_workflow(prompt, llm_provider="claude")
    
    # Display result
    print("=" * 60)
    print("WORKFLOW RESULT")
    print("=" * 60)
    print()
    print(result)
    print()
    print("=" * 60)
    print("âœ“ Workflow completed successfully!")
    print("=" * 60)

if __name__ == "__main__":
    main()
EOF

# Make executable
chmod +x simple_workflow.py
```

    """Calculate Fibonacci sequence up to n terms.

    Returns:
        List of Fibonacci numbers
    
    Raises:
        ValueError: If n is negative
